pipeline {
    agent any

    stages {

        stage('PULL') {
            steps {
                   git 'https://github.com/HarshCraft/secondstageproject-ekscluster.git'
            }
        }

        stage('PLAN') {
            steps {
                sh '''
                terraform init -reconfigure
                terraform plan
                '''
            }
        }

        stage('APPROVAL') {
            steps {
                timeout(time: 30, unit: 'MINUTES') {
                    input message: 'Approval is required. Shall we proceed?, OK : "Approve"'
                }
            }
        }

        stage('APPLY') {
            steps {
                sh '''
                terraform apply --auto-approve
                '''
            }
        }
	stage('KUBECTL CONFIG') {
    	steps {
        	sh '''
        	aws eks update-kubeconfig --region eu-west-1 --name my-cluster-19
        	'''
    		}
	}

    }
}
